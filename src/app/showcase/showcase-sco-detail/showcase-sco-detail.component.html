import { ScoCacheService } from 'sco-components-lib';
import { Component } from '@angular/core';

@Component({
  selector: 'app-showcase-sco-detail',
  templateUrl: './showcase-sco-detail.component.html',
  styleUrls: ['./showcase-sco-detail.component.scss']
})

export class ShowcaseScoDetailComponent {

  public showDetail:boolean;

  public showTitle:boolean;
  public showOverlay:boolean;
  public enableClickOutside:boolean;

  constructor(
    private cacheService: ScoCacheService
  ) { 
    this.cacheService.setElement("title", "Detail");
    this.showTitle = true;
    this.showOverlay = true;
    this.enableClickOutside = true;
  }
  
  openExample1() {
    this.showDetail = true;
    this.showTitle = true;
    this.showOverlay = true;
    this.enableClickOutside = true;
  }

  openExample2() {
    this.showDetail = true;
    this.showTitle = false;
    this.showOverlay = true;
    this.enableClickOutside = true;
  }

  openExample3() {
    this.showDetail = true;
    this.showTitle = true;
    this.showOverlay = false;
    this.enableClickOutside = true;
  }

  openExample4() {
    this.showDetail = true;
    this.showTitle = true;
    this.showOverlay = true;
    this.enableClickOutside = false;
  }

  closeDetail() {
    this.showDetail = false;
  }
}
Nombre Descripcción Tipo Valor por defecto Posibles valores
showOverlay Indica si se muestra o no el overlay del Detalle boolean true
showTitle Indica si se muestra o no el título del Detalle boolean true
clickOutsideEnabled Habilita o deshabilita la directiva click outside del detalle boolean true
closeOnClick Cierra la ventalla detalle al hacer un click en el contenido (Menus) boolean false
forceCloseNow Al cambiar la propiedad a true, cierra la ventana del detalle, hay que cambiarla a false antes de volver a abrirlo boolean false
Nombre Descripcción Valor devuelto
close Evento al cerrar el detalle boolean
