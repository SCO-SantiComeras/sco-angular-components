import { Component } from '@angular/core';
import { ScoCacheService } from 'sco-angular-components';

@Component({
  selector: 'app-showcase-sco-detail',
  templateUrl: './showcase-sco-detail.component.html',
  styleUrls: ['./showcase-sco-detail.component.scss']
})

export class ShowcaseScoDetailComponent {

  /* Detail Show Flag */
  public showDetail: boolean;

  /* Detail Input Params */
  public showOverlay: boolean = true;
  public showTitle: boolean = true;
  public clickOutsideEnabled: boolean = true;
  public closeOnClick: boolean = false;
  public forceCloseNow: boolean = false;
  public rightPosition: boolean = true;
  public showCloseButton: boolean = true;
  public alwaysOpen: boolean = false;
  public border: string = '1px solid';
  public borders: string[] = ['left', 'top', 'bottom', 'right'];

  constructor(private readonly cacheService: ScoCacheService) { 
    this.cacheService.setElement("title", "Detail");
  }
  
  openDetail() {
    this.showDetail = true;
  }

  closeDetail($event: boolean): void {
    if ($event) {
      this.showDetail = false;
    }
  }
}
Nombre Descripcción Tipo Valor por defecto Posibles valores
showOverlay Indica si se muestra o no el overlay del Detalle boolean true
showTitle Indica si se muestra o no el título del Detalle boolean true
clickOutsideEnabled Habilita o deshabilita la directiva click outside del detalle boolean true
closeOnClick Cierra la ventalla detalle al hacer un click en el contenido (Menus) boolean false
forceCloseNow Al cambiar la propiedad a true, cierra la ventana del detalle, hay que cambiarla a false antes de volver a abrirlo boolean false
rightPosition Indicador para mostrar el detalle a la derecha o a la izquierda boolean true
showCloseButton Indicador para mostrar o no el botón de cerrar de la cabecera boolean true
alwaysOpen Indicador para mostrar fijo el detalle boolean false
border Anchura y tipo de borde del detalle string '1px solid'
borders Bordes a los que se le aplica el tipo de borde string[] ['left', 'top', 'bottom', 'right']
  • 'left'
  • 'top'
  • 'bottom'
  • 'right'
Nombre Descripcción Valor devuelto
close Evento al cerrar el detalle boolean
