import { ScoCacheService, ScoToastService, ScoSelectItem } from 'sco-components-lib';
import { Component, OnInit } from '@angular/core';

@Component({
  selector: 'app-showcase-sco-dropdown',
  templateUrl: './showcase-sco-dropdown.component.html',
  styleUrls: ['./showcase-sco-dropdown.component.scss']
})

export class ShowcaseScoDropdownComponent implements OnInit {

  public options: ScoSelectItem[] = [];
  public valueEx2: any;
  public valueEx3: any;
  public valueEx5: any;

  constructor(
    private toastService: ScoToastService,
    private cacheService: ScoCacheService
  ) { 
    this.cacheService.setElement("title", "Dropdown");
  }

  ngOnInit() {
    this.options = [
      new ScoSelectItem("Label1", "Value1"),
      new ScoSelectItem("Label2", "Value2"),
      new ScoSelectItem("Label3", "Value3"),
      new ScoSelectItem("Label4", "Value4"),
    ];

    this.valueEx2 = "";
    this.valueEx3 = "";
    this.valueEx5 = "Value4";
  }

  select($event) {
    console.log($event);
    this.toastService.addInfoMessage("Éxito", `Has elegido: ${JSON.stringify($event)}`);
  }
}
Nombre Descripcción Tipo Valor por defecto Posibles valores
options Opciones a mostrar en el dropdown ScoSelectItem[] []
labelNoResults Label cuando no hay elementos string "No results"
label Label del dropdown string ""
showLabel Indicador para mostrar el label del dropdown boolean false
require Indica si es obligatorio el valor del dropdown boolean false
showFilter Indicador para mostrar o no el buscador boolean true
Nombre Descripcción Valor devuelto
selectItem Evento al seleccionar un item del dropdown ScoSelectItem
