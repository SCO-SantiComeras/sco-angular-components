import { Component, OnInit } from '@angular/core';
import { ScoCacheService } from 'projects/sco-components/src/public-api';

@Component({
  selector: 'app-indicator-showcase',
  templateUrl: './indicator-showcase.component.html',
  styleUrls: ['./indicator-showcase.component.scss']
})
export class IndicatorShowcaseComponent {

  value1: number;
  value2: number;
  value3: number;
  value4: number;
  value5: number;

  constructor(private cacheService: ScoCacheService) {
    this.cacheService.setElement('title', 'Indicator')
    this.value1 = 25;
    this.value2 = 50;
    this.value3 = 75;
    this.value4 = 100;
    this.value5 = 100;
  }

}

strokeWidth Modifica el borde del indicador en pixeles. number 14
showPercentage Muestra el porcentaje en el indicador boolean false
color Permite cambiar el color de la barra del indicador. Sino se indica coge la del theme. string undefined
