import { scoTranslateService, ScoCacheService, ScoToastService } from 'sco-angular-components';
import { Component, OnInit } from '@angular/core';

@Component({
  selector: 'app-showcase-sco-file-uploader',
  templateUrl: './showcase-sco-file-uploader.component.html',
  styleUrls: ['./showcase-sco-file-uploader.component.scss']
})
export class ShowcaseScoFileUploaderComponent implements OnInit {

  public openAccordionOptions: boolean;

  public showFileUploader: boolean;

  public showLabel: boolean;
  public required: boolean;
  public label: string;
  public showBorder: boolean;
  public borderColor: string;
  public showUploadBtn: boolean;
  public labelUploadBtn: string;
  public showCleanBtn: boolean;
  public labelCleanBtn: string;
  public base64: boolean;
  public labelBase64Title: string;
  public base64TextRows: number;

  constructor(
    private readonly toastService: ScoToastService,
    private readonly cacheService: ScoCacheService,
    public readonly translateService: scoTranslateService,
  ) { 
    this.cacheService.setElement("title", "File Uploader");

    this.openAccordionOptions = true;

    this.showFileUploader = false;

    this.showLabel = false;
    this.required = false;
    this.label = 'File uploader';
    this.showBorder = false;
    this.borderColor = '#000';
    this.showUploadBtn = true;
    this.labelUploadBtn = 'Subir';
    this.showCleanBtn = true;
    this.labelCleanBtn = 'Limpiar';
    this.base64 = false;
    this.labelBase64Title = 'Base64';
    this.base64TextRows = 4;
  }

  ngOnInit() {
    this.openAccordionOptions = true;

    this.showFileUploader = true;

    this.showBorder = false;
    this.borderColor = '#000';
    this.showUploadBtn = true;
    this.labelUploadBtn = 'Subir';
    this.showCleanBtn = true;
    this.labelCleanBtn = 'Limpiar';
    this.base64 = false;
    this.labelBase64Title = 'Base64';
    this.base64TextRows = 4;
  }

  onCLickAccordionOptions() {
    this.openAccordionOptions = !this.openAccordionOptions;
  }

  openFileUploader() {
    this.showFileUploader = true;
  }

  /* Component Functions */
  uploadBtn($event) {
    this.toastService.addSuccessMessage(
      "Success",
      `Botón subir pulsado: ${JSON.stringify($event ? $event.name : null)}`
    );
  }

  cleanBtn($event) {
    this.toastService.addSuccessMessage(
      "Success",
      "Botón limpiar pulsado"
    );
  }

  copyBtn($event) {
    this.toastService.addSuccessMessage(
      "Success",
      `Botón copiar pulsado: ${$event[0]} - ${$event[1]}`
    );
  }
}
Nombre Descripcción Tipo Valor por defecto Posibles valores
label Texto del componente string 'Seleccione imagen'
showLabel Indicador para mostrar el texto del componente boolean false
required Indicador para mostrar el texto de elemento obligatorio boolean false
showBorder Indicador para mostrar el borde o no del componente boolean false
borderColor Color del borde del componente (Hex, name) string '#000'
showUploadBtn Indicador para mostrar el botón de subida de archivo boolean true
labelUploadBtn Texto del botón de subida de archivo string 'Subir'
showCleanBtn Indicador para mostrar el botón de limpar archivo boolean true
labelCleanBtn Texto del botón de limpiar archivo string 'Limpiar'
base64 Indicador para mostrar el decodificador en base64 boolean false
labelBase64Title Texto del título del decodificador base64 string 'Base64'
base64TextRows Número de filas del textarea del decodificar de base64 number 4
Nombre Descripcción Valor devuelto
uploadBtn Evento al pulsar el botón de subida de archivo File
cleanBtn Evento al pulsar el botón de limpieza de archivo boolean
copyBtn Evento al pulsar el botón de copiar valor de un decodificar string['editor', 'valor']
