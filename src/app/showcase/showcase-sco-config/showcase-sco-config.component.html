import { ScoCacheService, scoConfigService } from 'sco-components-lib';
import { Component } from '@angular/core';

@Component({
  selector: 'app-showcase-sco-config',
  templateUrl: './showcase-sco-config.component.html',
  styleUrls: ['./showcase-sco-config.component.scss']
})

export class ShowcasescoConfigComponent {

  constructor(
    private scoConfig: scoConfigService,
    private cacheService: ScoCacheService
  ) { 
    this.cacheService.setElement("title", "Config Service");
    console.log("All Data: " + this.scoConfig.getAllData());
    console.log("Variable2: " + this.scoConfig.getData('variable2'));
  }
}
-- App.Module 

import { APP_INITIALIZER, NgModule } from '@angular/core';
import { BrowserModule, } from '@angular/platform-browser';
import { BrowserAnimationsModule } from '@angular/platform-browser/animations';
import { ScoComponentsLibModule } from 'sco-components-lib';
import { scoConfigService } from 'sco-components-lib';
                    
export function configFactory(provider: scoConfigService) {
    return () => provider.getDataFromJson('assets/config/data.json');
}

@NgModule({
    imports: [
        BrowserModule,
        BrowserAnimationsModule,
        ScoComponentsLibModule,
    ],
    providers: [
        scoConfigService, 
        {
            provide: APP_INITIALIZER,
            useFactory: configFactory,
            deps: [
                scoConfigService
                ],
            multi: true
        }
    ],
    bootstrap: [AppComponent]
})

export class AppModule { }
