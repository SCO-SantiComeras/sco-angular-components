import { animate, state, style, transition, trigger } from '@angular/animations';
import { Component } from '@angular/core';
import { ScoAction, ScoCacheService, ScoToastService } from 'projects/sco-components/src/public-api';

@Component({
  selector: 'app-split-button-showcase',
  templateUrl: './split-button-showcase.component.html',
  styleUrls: ['./split-button-showcase.component.scss']
})
export class SplitButtonShowcaseComponent {

  public actionsEx1: ScoAction<string>[] = [
    { label: 'Save', value: 'SAVE', item: 'save', icon: 'fa fa-floppy-o'},
    { label: 'Update', value: 'UPDATE', item: 'update', icon: 'fa fa-refresh'},
    { label: 'Delete', value: 'DELETE', item: 'delete', icon: 'fa fa-trash'},
  ];

  constructor(
    public cacheService: ScoCacheService,
    public toastService: ScoToastService
  ) { 
    this.cacheService.setElement('title', 'Split-button');
  }
  
  selectActionEx1($event){
    this.toastService.addSuccessMessage("Accion seleccionada ejemplo 1", JSON.stringify($event));
  }
}
