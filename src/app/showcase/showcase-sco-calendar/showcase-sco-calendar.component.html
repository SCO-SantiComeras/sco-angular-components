import { Component, ViewEncapsulation } from '@angular/core';
import { ScoToastService } from 'projects/sco-components/src/public-api';
import { ScoCacheService } from 'projects/sco-components/src/services/sco-cache.service';
import { ScoConstantsService } from 'projects/sco-components/src/services/sco-constants.service';

@Component({
  selector: 'app-calendar-showcase',
  templateUrl: './calendar-showcase.component.html',
  styleUrls: ['./calendar-showcase.component.scss'],
  encapsulation: ViewEncapsulation.None
})
export class CalendarShowcaseComponent {

  value1: Date;
  value2: Date;
  value3: Date = new Date(2021, 0, 1);
  value4: Date[];
  value5: Date[];
  value6: Date;
  value7: Date = new Date();
  value8: Date = new Date();

  constructor(
    public constants: ScoConstantsService, 
    public cacheService: ScoCacheService,
    public toasatService: ScoToastService
  ) { 
    this.cacheService.setElement('title', 'Calendar');
    this.value7.setHours(8,30);
  }
  
  selectEx2($event: Date){
    console.log("Fecha elegida: " , $event);
    this.toasatService.addSuccessMessage("Seleccion fecha", "Fecha elegida: " + $event);
  }

  selectEx4($event: Date[]){
    console.log("Fecha/s elegida/s: " , $event);
    console.log("Fecha/s elegida/s en el value4: " , this.value4);
    this.toasatService.addSuccessMessage("Seleccion fecha", "Fecha/s elegida/s: " + $event);
    this.toasatService.addSuccessMessage("Seleccion fecha", "Fecha/s elegida/s en el value4: " + this.value4);
  }

  selectEx5($event: Date[]){
    console.log("Fecha/s elegida/s: " , $event);
    console.log("Fecha/s elegida/s en el value5: " , this.value5);
    this.toasatService.addSuccessMessage("Seleccion fecha", "Rango de fechas: " + $event);
    this.toasatService.addSuccessMessage("Seleccion fecha", "Fecha/s elegida/s en el value5: " + this.value5);
  }

  selectEx6($event: Date){
    console.log("Fecha y tiempo elegida: " , $event);
    this.toasatService.addSuccessMessage("Seleccion fecha", "Fecha y tiempo elegida: " + $event);
  }

  selectEx7($event: Date){
    console.log("Tiempo elegido: " , $event);
    this.toasatService.addSuccessMessage("Seleccion fecha", "Tiempo elegido: " + $event);
  }

  clearClick($event: MouseEvent){
    console.log($event);
    this.toasatService.addSuccessMessage("Limpiar fecha", "Fecha limpiada");
  }
}

maxDate Fecha maxima a elegir Date undefined
disabled Indica si esta o no deshabilitado boolean false
readonlyInput Indica si o no de solo lectura boolean false
inputStyleClass Clases adicionales a añadir string ''
dateFormat Formato de la fecha seleccionada string dd/mm/yy
hourFormat Formato de la hora string 24
stepHour Incremento/decremento de la hora number 1
stepHour Incremento/decremento de la hora number 1
stepMinute Incremento/decremento de los minutos number 1
showButtonBar Indica si muestra la barra de botones del calendario boolean false
selectionMode Indica el modo de seleccion string ScoCalendarConstants.SELECTION_SINGLE
  • ScoCalendarConstants.SELECTION_SINGLE
  • ScoCalendarConstants.SELECTION_MULTIPLE
  • ScoCalendarConstants.SELECTION_RANGE
inputId Indica el id del calendar string ''
name Indica el name del calendar string ''
language Indica el idioma del calendario string LocaleConstants.ES
  • LocaleConstants.ES
  • LocaleConstants.EN
touchUI Indicador para mostrar el calendario en modo popup boolean false
placeholder Texto que aparece mientras no hay una fecha seleccionada string undefined
