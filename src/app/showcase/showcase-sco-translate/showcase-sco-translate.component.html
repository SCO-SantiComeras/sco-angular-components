-- Showcase 

import { ScoToastService, ScoCacheService, scoTranslateService } from 'sco-angular-components';
import { Component, OnInit } from '@angular/core';

@Component({
  selector: 'app-showcase-sco-translate',
  templateUrl: './showcase-sco-translate.component.html',
  styleUrls: ['./showcase-sco-translate.component.scss']
})

export class ShowcasescoTranslateComponent implements OnInit {

  constructor(
    private cacheService: ScoCacheService,
    private translateService: scoTranslateService,
    private toastService: ScoToastService
  ) { 
    this.cacheService.setElement("title", "Translate Service");
  }

  ngOnInit() {
    this.toastService.addInfoMessage("label.example.1", this.translateService.getTranslate('label.example.1'));
    this.toastService.addInfoMessage("label.example.2", this.translateService.getTranslate('label.example.2'));
    console.log(this.translateService.getTranslate('label.example.1'));
    console.log(this.translateService.getTranslate('label.example.2'));
  }
}

-- App.Module 

import { APP_INITIALIZER, NgModule } from '@angular/core';
import { BrowserModule, } from '@angular/platform-browser';
import { BrowserAnimationsModule } from '@angular/platform-browser/animations';
import { ScoComponentsLibModule } from 'sco-angular-components';
import { scoTranslateService } from 'sco-angular-components';
                    
export function translateFactory(provider: scoTranslateService) {
    return () => provider.getData('assets/i18n/');
}

@NgModule({
    imports: [
        BrowserModule,
        BrowserAnimationsModule,
        ScoComponentsLibModule,
    ],
    providers: [
        scoTranslateService, 
        {
          provide: APP_INITIALIZER,
          useFactory: translateFactory,
          deps: [
              scoTranslateService
          ],
          multi: true
        }
    ],
    bootstrap: [AppComponent]
})

export class AppModule { }
