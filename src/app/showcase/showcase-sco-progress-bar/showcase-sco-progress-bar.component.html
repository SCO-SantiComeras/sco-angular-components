  selector: 'app-progress-bar-showcase',
  templateUrl: './progress-bar-showcase.component.html',
  styleUrls: ['./progress-bar-showcase.component.scss']
})
export class ProgressBarShowcaseComponent {

  public valueEx1 = 0;
  public valueEx2 = 0;
  public valueEx3 = 0;
  public valueEx4 = 0;
  public valueEx5 = 0;

  constructor(private cacaheService: ScoCacheService) { 
    this.cacaheService.setElement('title', 'Progress bar');
  }

  increaseEx1(){
    if(this.valueEx1 + 10 > 100){
      this.valueEx1 = 100;
    }else{
      this.valueEx1 = this.valueEx1 + 10;
    }
  }

  decreaseEx1(){
    if(this.valueEx1 - 10 < 0){
      this.valueEx1 = 0;
    }else{
      this.valueEx1 = this.valueEx1 - 10;
    }
  }

Nombre Descripción Tipo Valor por defecto Posibles valores (constants)
value Indica el porcentaje de la barra de progreso. Entre 0 y 100. number undefined
label Indica el texto que dentro de la barra de progreso string undefined
height Altura de la barra de progreso. number 16
striped Estilo con lineas en la barra de progreso. boolean false
animated Indica si anima la barra de progreso. Es necesario que striped este activo boolean false
Nombre Descripcción Valor devuelto
loaded Evento al terminar de cargar la barra de progreso boolean
